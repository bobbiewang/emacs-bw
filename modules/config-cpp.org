* CC Mode

#+BEGIN_SRC emacs-lisp
  (defun bw/c++-mode-hook()
    (c-set-style "stroustrup")
    (c-set-offset 'inline-open 0))

  (add-hook 'c++-mode-hook 'bw/c++-mode-hook)
#+END_SRC
